public class PracticeMenu {
    
    public static void main(String[] mainArgs) {
        final int DISPLAY_HELLO = 1;
        final int DISPLAY_PI = 2;
        final int DISPLAY_SUB_MENU_01 = 3;
        final int EXIT = 4;
        
        int menuSelection = 0; 
        do {
            displayMainMenu();
            menuSelection = getInteger("Enter your Selection");    
            
            switch (menuSelection) {
                case DISPLAY_HELLO: 
                    displayHello();
                    break;
                case DISPLAY_PI:
                    displayPI();
                    break;
                case DISPLAY_SUB_MENU_01:
                	processSubMenu();
                	break;
                case EXIT:
                    break;
                default:
                    System.err.println("Invalid menu number. Please try again."); 
                    break;
            }
        } 
        while (menuSelection != EXIT);
        System.out.println("Goodbye!");
    }
    
    /////////////////////////////////////////////////////////////////////////////////////////////////////////////////
    
    private static void displayMainMenu() {
        System.out.println("");
        System.out.println("****************************************");
        System.err.println("          Main Menu         ");
        System.out.println("****************************************");
        System.out.println("       1. Display Hello");
        System.out.println("       2. Display PI");
        System.out.println("       3. Display Sub Menu");
        System.out.println("       4. Exit");
        System.out.println("");
    }
    
    private static void displayHello () {
        System.out.println("Hello");        
    }
    private static void displayPI () {
        System.out.println("pi = 3.1415 etc...");        
    }
    
    //------------------------------------------------------------------------
    
    private static void processSubMenu(){
    	
        final int SUB_MENU_DISPLAY_NAME = 1;
        final int SUB_MENU_DISPLAY_RANDOM_NUMBER = 2;
        final int SUB_MENU_EXIT = 3;
    	
        int subMenuSelection = 0;
        
    	displaySubMenu();
        subMenuSelection = getInteger("Enter your Selection");    
        
        switch (subMenuSelection) {
            case SUB_MENU_DISPLAY_NAME: 
                displayName();
                processSubMenu();
                break;
            case SUB_MENU_DISPLAY_RANDOM_NUMBER:
                displayRandomNumber();
                processSubMenu();
                break;
            case SUB_MENU_EXIT:
                break;
            default:
                System.err.println("Invalid menu number. Please try again."); 
                break;
        }
  
        //copy code to here
        //on option 3, return to the main menu
    }

    //add sub menu methods here
    
    private static void displaySubMenu() {
        System.out.println("");
        System.out.println("****************************************");
        System.err.println("          SubMenu Options ");
        System.out.println("****************************************");
        System.out.println("       1. Display a Name ");
        System.out.println("       2. Display a random number");
        System.out.println("       3. Return to Main Menu");
        System.out.println("");
    }
    
    private static void displayName() {
    	String name = getString("What is your name? ");
    	System.out.println(name);
    }
    
    private static void displayRandomNumber() {
    	int randomNumber = getRandomNumber(0, 100);
    	System.out.println(randomNumber);
    }
	
}
